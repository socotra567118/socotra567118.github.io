| &#37; | &#38;#37; | &#38; | &#38;#38; |
| &#39; | &#38;#39; | &#40; | &#38;#40; |
| &#41; | &#38;#41; | &#42; | &#38;#42; |
| &#43; | &#38;#43; | &#44; | &#38;#44; |
| &#45; | &#38;#45; | &#46; | &#38;#46; |
| &#47; | &#38;#47; | &#58; | &#38;#58; |
| &#59; | &#38;#59; | &#92; | &#38;#92; |
| &#96; | &#38;#96; |
